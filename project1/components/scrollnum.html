# 滚动的Num

滚动的Num

# 基础用法

<template>
    <div>
        <xw-scroll-num :seriesData="seriesData" />
    </div>
</template>
<script>
    /*
  * name: 滚动的Num
  * Author: xc9010
  * type: show
  * img: /img/scrollnum1.png
  * desc: 基础用法
  * src: /components/scrollnum.html
  * */
  import XwScrollNum from '../../../../src/packages/ScrollNum/src/scrollnum'

  export default {
    components: { XwScrollNum },
    data () {
      return {
        seriesData: {
          name: '',
          value: 56789,
        }
      }
    },
    mounted() {
      this.timeout()
    },
    methods: {
      timeout() {
        const self = this
        setInterval(() => {
          self.seriesData.value = self.seriesData.value + 1
        }, 2000)
      }
    }
  }
</script>

# 属性

styles Object 数字样式
titleStyle Object 标题样式
