# 鼠标绘制点线面

head 引入
<head>
    <script type='text/javascript' src='http://api.map.baidu.com/library/DrawingManager/1.4/src/DrawingManager_min.js'></script>  
    <link rel='stylesheet' href='//api.map.baidu.com/library/DrawingManager/1.4/src/DrawingManager_min.css'>
</head>

<template>
    <div style="width: 100%;height: 500px">
        <xw-bmap
                open-mouse-draw
                @onCreate="getRes"
                :draw-style="drawStyle"
        />
    </div>
</template>
<script>
    /*
* name: 百度地图
* Author: xc9010
* type: map
* img: /img/bmap4.png
* desc: 支持绘制
* src: /components/bmap.html
* */
  import XwBmap from '../../../../src/packages/BMap/src/BMap'

  export default {
    components: { XwBmap },
    data () {
      return {
        drawStyle: {
          strokeColor:"red",    //边线颜色。
          fillColor:"red",      //填充颜色。当参数为空时，圆形将没有填充效果。
          strokeWeight: 3,       //边线的宽度，以像素为单位。
          strokeOpacity: 0.8,	   //边线透明度，取值范围0 - 1。
          fillOpacity: 0.6,      //填充的透明度，取值范围0 - 1。
          strokeStyle: 'solid' //边线的样式，solid或dashed。
        }
      }
    },
    methods: {
      getRes(val) {
        console.log(val)
      }
    }
  }
</script>

# 属性

参数 类型 默认值 是否必填 说明 可选值/参考值
className String 样式类
centerPoint Object { x: '121.491', y: '31.233' } 中心点坐标
zoom Number 11 地图级别
MapOptions Object { enableMapClick: false } 地图配置
scrollWheel Boolean true 鼠标滚轮缩放
navigation Object 平移缩放控件 参考百度API ControlAnchor (opens new window)
anchor Object 比例尺 参考百度API ControlAnchor (opens new window)
points Array 画点数据 [{ lng: 121.491, lat: 31.233 },{ lng: 121.495, lat: 31.233 },{ lng: 121.498, lat: 31.233 },]
label Array 文本标注 [{ txt: 'label1', lng: 121.491, lat: 31.233 },{ txt: 'label2', lng: 121.595, lat: 31.233 },{ txt: 'label3', lng: 121.398, lat: 31.233 }]
labelStyle Object { padding: '0 5px', fontSize : "12px", height : "20px", lineHeight : "20px", fontFamily:"微软雅黑" } 文本标注样式
openMouseDraw Boolean false 开启鼠标画点线面辅助工具，需要head引入js/css
drawStyle Object 覆盖物的样式，参考 百度API (opens new window)
onCreate Function 返回所画出的点线面对象{str: String格式，obj:Object格式}
openHeatMap Boolean 热力图开关
heatMapOption Object 热力图配置信息
colorGradient Object 热力图-热力点颜色
heatMapData Array 热力图数据
boundary String 指定行政区高亮
