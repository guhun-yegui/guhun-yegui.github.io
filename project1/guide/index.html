# 开发指南-组件部分

开发规范

# 1，确保只在 beforeMount 或者 mounted 访问浏览器 / DOM 的 API

# 下载安装项目

    git clone http://gitlab.xwsoftlan.com/xw_components/xwui.git
    cd xwui
    npm install

TIP

# scripts命令如下表格

命令 说明 备注
npm run docs:dev 启动本地 端口修改路径->./docs/config.js
npm run docs:build 构建文档产物 目标路径->./public
npm run doc:clean 删除文档产物 目标路径->./public
npm run docs:build-list 构建组件库文档汇总 目标路径->./config
npm run lib:build 构建npm包 目标路径->./lib
npm run lib:clean 删除npm包 目标路径->./lib
npm run build:utils 构建npm包资源到lib 目标路径->./lib
npm run lib:publish 发布npm包 发布前请修改package.json/version版本号

# 新增组件

    cd src/packages
    mkdir Button

新增目录结构如下

    .
    ├─ src
    │  └─ button.vue
    └─ index.js

示例button.vue

    <template>
        <button>按钮1</button>
    </template>
    
    <script>
        export default {
          name: 'XwButton' // 定义组件名，统一命名规范为Xw+**
        }
    </script>

示例index.js

import Button from './src/button';

/* istanbul ignore next */
Button.install = function(Vue) {
  Vue.component(Button.name, Button);
};

export default Button;

# 修改packages/index.js



 






 


















 




import { Bar, Pie, Line } from './echars'
import Button from './Button'
// 1,这里引入新增的组件

const components = [
  Bar,
  Pie,
  Line,
  Button,
  // 2,引入新增的组件
];

const install = (Vue) => {
  components.forEach(component => {
    Vue.component(component.name, component)
  })
};

if (typeof window !== 'undefined' && window.Vue) {
  install(window.Vue);
}

export default {
  install,
  Bar,
  Button,
  Pie,
  Line,
  // 3,导出新增的组件
}
